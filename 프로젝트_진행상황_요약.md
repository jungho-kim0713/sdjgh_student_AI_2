def get_persona_from_db(role_key):
    """DB에서 페르소나 조회 (활성화된 것만)"""
    return PersonaDefinition.query.filter_by(role_key=role_key, is_active=True).first()

def get_system_prompt_from_db(persona_id, provider):
    """DB에서 시스템 프롬프트 조회 (provider별, fallback to default)"""
    prompt = PersonaSystemPrompt.query.filter_by(persona_id=persona_id, provider=provider).first()
    if prompt:
        return prompt.system_prompt
    default_prompt = PersonaSystemPrompt.query.filter_by(persona_id=persona_id, provider="default").first()
    return default_prompt.system_prompt if default_prompt else ""
```

// 추가된 3개 함수

let currentPrompts = {}; // provider별 프롬프트 캐시

async function loadAllPrompts(personaId) {
    const response = await fetch(`/api/admin/persona/${personaId}/prompts`);
    const data = await response.json();
    currentPrompts = {};
    if (data.prompts) {
        data.prompts.forEach(p => {
            currentPrompts[p.provider] = p.system_prompt;
        });
    }
    loadPromptForProvider();
}

function loadPromptForProvider() {
    const provider = document.getElementById('promptProvider').value;
    const textarea = document.getElementById('systemPrompt');
    textarea.value = currentPrompts[provider] || '';
}

async function saveSystemPrompt() {
    if (!selectedPersonaId) return alert('페르소나를 먼저 선택하세요');
    const provider = document.getElementById('promptProvider').value;
    const prompt = document.getElementById('systemPrompt').value.trim();
    if (!prompt) return alert('프롬프트를 입력하세요');

    const response = await fetch(`/api/admin/persona/${selectedPersonaId}/prompt`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ provider, system_prompt: prompt })
    });
    const result = await response.json();
    if (result.success) {
        alert('시스템 프롬프트가 저장되었습니다');
        currentPrompts[provider] = prompt;
    }
}
```

#### routes/admin_persona.py (lines 531-610)
```python
# 2개 API 엔드포인트 추가

@admin_persona_bp.route("/api/admin/persona/<int:persona_id>/prompts", methods=["GET"])
@admin_required
def get_persona_prompts(persona_id):
    """특정 페르소나의 모든 프롬프트 조회"""
    prompts = PersonaSystemPrompt.query.filter_by(persona_id=persona_id).all()
    return jsonify({
        "prompts": [{"provider": p.provider, "system_prompt": p.system_prompt} for p in prompts]
    })

@admin_persona_bp.route("/api/admin/persona/<int:persona_id>/prompt", methods=["PUT"])
@admin_required
def update_persona_prompt(persona_id):
    """시스템 프롬프트 업데이트 또는 생성"""
    persona = db.session.get(PersonaDefinition, persona_id)
    if not persona:
        return jsonify({"error": "페르소나를 찾을 수 없습니다"}), 404

    data = request.json
    provider = data.get("provider", "default")
    system_prompt = data.get("system_prompt", "")

    prompt = PersonaSystemPrompt.query.filter_by(
        persona_id=persona_id,
        provider=provider
    ).first()

    if prompt:
        # 기존 프롬프트 업데이트
        prompt.system_prompt = system_prompt
        prompt.updated_by = current_user.id
        prompt.updated_at = datetime.datetime.utcnow()
    else:
        # 새 프롬프트 생성
        prompt = PersonaSystemPrompt(
            persona_id=persona_id,
            provider=provider,
            system_prompt=system_prompt,
            updated_by=current_user.id
        )
        db.session.add(prompt)

    db.session.commit()
    return jsonify({"success": True})
```

## 💡 중요 참고사항

### DB 구조
```
PersonaDefinition (페르소나 메타데이터)
├── id (PK)
├── role_key (고유 식별자)
├── role_name (표시명)
├── is_system (시스템 페르소나 여부)
├── is_active (활성화 상태)
├── allow_user (일반 사용자 허용)
├── allow_teacher (교사 허용)
├── use_rag (RAG 사용 여부)
└── AI 모델 기본 설정 (default_provider, default_model, temperature 등)

PersonaSystemPrompt (페르소나별 시스템 프롬프트)
├── id (PK)
├── persona_id (FK → PersonaDefinition)
├── provider (default/openai/anthropic/google)
├── system_prompt (프롬프트 텍스트)
├── updated_by (수정자)
└── updated_at (수정 시각)
```
